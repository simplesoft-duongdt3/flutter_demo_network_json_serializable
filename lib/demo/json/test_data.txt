class Autogenerated {
  int _page;
  int _perPage;
  int _total;
  int _totalPages;
  Author _author;
  List<Data> _data;

  Autogenerated(
      {int page,
      int perPage,
      int total,
      int totalPages,
      Author author,
      List<Data> data}) {
    this._page = page;
    this._perPage = perPage;
    this._total = total;
    this._totalPages = totalPages;
    this._author = author;
    this._data = data;
  }

  int get page => _page;
  set page(int page) => _page = page;
  int get perPage => _perPage;
  set perPage(int perPage) => _perPage = perPage;
  int get total => _total;
  set total(int total) => _total = total;
  int get totalPages => _totalPages;
  set totalPages(int totalPages) => _totalPages = totalPages;
  Author get author => _author;
  set author(Author author) => _author = author;
  List<Data> get data => _data;
  set data(List<Data> data) => _data = data;

  Autogenerated.fromJson(Map<String, dynamic> json) {
    _page = json['page'];
    _perPage = json['per_page'];
    _total = json['total'];
    _totalPages = json['total_pages'];
    _author =
        json['author'] != null ? new Author.fromJson(json['author']) : null;
    if (json['data'] != null) {
      _data = new List<Data>();
      json['data'].forEach((v) {
        _data.add(new Data.fromJson(v));
      });
    }
  }

  Map<String, dynamic> toJson() {
    final Map<String, dynamic> data = new Map<String, dynamic>();
    data['page'] = this._page;
    data['per_page'] = this._perPage;
    data['total'] = this._total;
    data['total_pages'] = this._totalPages;
    if (this._author != null) {
      data['author'] = this._author.toJson();
    }
    if (this._data != null) {
      data['data'] = this._data.map((v) => v.toJson()).toList();
    }
    return data;
  }
}

class Author {
  String _firstName;
  String _lastName;

  Author({String firstName, String lastName}) {
    this._firstName = firstName;
    this._lastName = lastName;
  }

  String get firstName => _firstName;
  set firstName(String firstName) => _firstName = firstName;
  String get lastName => _lastName;
  set lastName(String lastName) => _lastName = lastName;

  Author.fromJson(Map<String, dynamic> json) {
    _firstName = json['first_name'];
    _lastName = json['last_name'];
  }

  Map<String, dynamic> toJson() {
    final Map<String, dynamic> data = new Map<String, dynamic>();
    data['first_name'] = this._firstName;
    data['last_name'] = this._lastName;
    return data;
  }
}

class Data {
  int _id;
  String _firstName;
  String _lastName;
  String _avatar;
  List<Images> _images;

  Data(
      {int id,
      String firstName,
      String lastName,
      String avatar,
      List<Images> images}) {
    this._id = id;
    this._firstName = firstName;
    this._lastName = lastName;
    this._avatar = avatar;
    this._images = images;
  }

  int get id => _id;
  set id(int id) => _id = id;
  String get firstName => _firstName;
  set firstName(String firstName) => _firstName = firstName;
  String get lastName => _lastName;
  set lastName(String lastName) => _lastName = lastName;
  String get avatar => _avatar;
  set avatar(String avatar) => _avatar = avatar;
  List<Images> get images => _images;
  set images(List<Images> images) => _images = images;

  Data.fromJson(Map<String, dynamic> json) {
    _id = json['id'];
    _firstName = json['first_name'];
    _lastName = json['last_name'];
    _avatar = json['avatar'];
    if (json['images'] != null) {
      _images = new List<Images>();
      json['images'].forEach((v) {
        _images.add(new Images.fromJson(v));
      });
    }
  }

  Map<String, dynamic> toJson() {
    final Map<String, dynamic> data = new Map<String, dynamic>();
    data['id'] = this._id;
    data['first_name'] = this._firstName;
    data['last_name'] = this._lastName;
    data['avatar'] = this._avatar;
    if (this._images != null) {
      data['images'] = this._images.map((v) => v.toJson()).toList();
    }
    return data;
  }
}
